## Goal

Tegakkan **test foundation minimum** sebelum Sprint 3 page port (Spec #7+) dimulai. Tanpa test, refactor masa depan = pray. Setelah ini, setiap pull request CI gate akan run typecheck + lint + test sebelum merge.

Setelah spec ini selesai:
- `pnpm test` jalan dengan minimum **8 test** hijau
- `pnpm test:ui` buka Vitest UI dashboard untuk dev experience
- `pnpm test:cov` generate coverage report (HTML + lcov)
- GitHub Actions workflow run test pada setiap push/PR ke `main`
- Pre-commit hook (optional) run test pada file modified saja

## Konteks

Per audit `docs/AUDIT_PRE_BETA_2026-05-01.md` Section A5: testing scored **1/10** (zero test). Risk register: refactor masa depan high-risk. Spec ini mitigate risk itu **sebelum** Sprint 3 page port menambah ~20+ component baru yang akan harder to test retroactively.

Pilihan framework: **Vitest** (bukan Jest). Reasoning:
- Vite-powered → 10x lebih cepat dari Jest
- TypeScript native (gak perlu babel/swc)
- API mirror Jest (`describe/it/expect`) — easy migration kalau perlu
- Built-in coverage via c8 / istanbul
- Watch mode + UI dashboard built-in

### 5. Tulis 8 test minimum (di `apps/web/src/__tests__/`)

Test file conventions:
- Filename `<source-name>.test.ts` atau `.test.tsx` (kalau JSX)
- Pakai `describe` + `it` (atau `test`) dari `vitest`
- Import dengan path alias `@/` atau `@jw/data/`
- Setiap test independen (gak share state)

Contoh skeleton untuk **`lib-format.test.ts`**:

```ts
import { describe, it, expect } from 'vitest';
import { formatNumber, formatDate, formatCurrency } from '@/lib/format';

describe('formatNumber', () => {
  it('formats integer with thousand separator (Indonesian locale)', () => {
    expect(formatNumber(31990)).toBe('31.990');
  });
  
  it('handles zero', () => {
    expect(formatNumber(0)).toBe('0');
  });
});

describe('formatDate', () => {
  it('formats ISO string to Indonesian date', () => {
    const result = formatDate('2026-05-01T00:00:00Z');
    expect(result).toContain('Mei');
    expect(result).toContain('2026');
  });
});

describe('formatCurrency', () => {
  it('formats Rupiah with thousand separator', () => {
    expect(formatCurrency(150000)).toBe('Rp 150.000');
  });
});
```

Contoh skeleton untuk **`lib-nala-store.test.ts`**:

```ts
import { describe, it, expect, beforeEach } from 'vitest';
import { useNalaStore } from '@/lib/nala/store';

describe('useNalaStore', () => {
  beforeEach(() => {
    useNalaStore.getState().clearChat();
    useNalaStore.getState().closePanel();
  });

  it('opens panel with optional context', () => {
    const { openPanel } = useNalaStore.getState();
    openPanel('tentang halaman ini');
    const state = useNalaStore.getState();
    expect(state.isPanelOpen).toBe(true);
    expect(state.currentContext).toBe('tentang halaman ini');
  });

  it('addMessage assigns id and createdAt automatically', () => {
    const { addMessage } = useNalaStore.getState();
    addMessage({ role: 'user', content: 'halo' });
    const state = useNalaStore.getState();
    expect(state.messages).toHaveLength(1);
    expect(state.messages[0]).toMatchObject({
      role: 'user',
      content: 'halo',
    });
    expect(state.messages[0].id).toBeTruthy();
    expect(state.messages[0].createdAt).toBeTruthy();
  });

  it('clearChat empties messages array', () => {
    const { addMessage, clearChat } = useNalaStore.getState();
    addMessage({ role: 'user', content: 'a' });
    addMessage({ role: 'nala', content: 'b' });
    clearChat();
    expect(useNalaStore.getState().messages).toEqual([]);
  });
});
```

import { describe, it, expect } from 'vitest';
import { getMockResponse } from '@/lib/nala/mock-responses';

describe('getMockResponse', () => {
  it('matches keyword pasal 28E (case insensitive)', () => {
    const result = getMockResponse('Apa itu pasal 28E?');
    expect(result.content).toContain('Pasal 28E');
    expect(result.citations).toHaveLength(1);
    expect(result.citations[0]).toMatchObject({ index: 1 });
  });

  it('matches keyword kelas online', () => {
    const result = getMockResponse('Saya cari kelas online publik');
    expect(result.content).toContain('Kelas online');
  });

  it('matches keyword opini', () => {
    const result = getMockResponse('Bantu draft opini');
    expect(result.content).toContain('Opini editorial');
  });

  it('returns fallback for unknown topic', () => {
    const result = getMockResponse('Cuaca hari ini gimana?');
    expect(result.content).toContain('mode beta');
    expect(result.citations).toEqual([]);
  });
});

import { describe, it, expect, beforeEach } from 'vitest';
import { render, screen } from '@testing-library/react';
import { NalaPanel } from '@/components/nala/nala-panel';
import { useNalaStore } from '@/lib/nala/store';

describe('<NalaPanel />', () => {
  beforeEach(() => {
    useNalaStore.getState().clearChat();
    useNalaStore.getState().closePanel();
  });

  it('renders empty state when no messages', () => {
    useNalaStore.getState().openPanel();
    render(<NalaPanel />);
    expect(screen.getByText(/Halo! Mau ngobrolin apa hari ini?/i)).toBeInTheDocument();
    expect(screen.getByLabelText(/Tutup panel Nala/i)).toBeInTheDocument();
  });

  it('renders messages when chat has history', () => {
    useNalaStore.getState().openPanel();
    useNalaStore.getState().addMessage({ role: 'user', content: 'Halo Nala' });
    render(<NalaPanel />);
    expect(screen.getByText('Halo Nala')).toBeInTheDocument();
  });
});

### 6. GitHub Actions workflow

`.github/workflows/ci-test.yml`:

```yaml
name: CI — typecheck + lint + test

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  ci:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4
        with:
          version: 9.12.0

      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: 'pnpm'

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Typecheck
        run: pnpm --filter @jw/web typecheck

      - name: Lint
        run: pnpm --filter @jw/web lint

      - name: Test
        run: pnpm --filter @jw/web test
```
